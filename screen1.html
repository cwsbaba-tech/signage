// ========================================== // 【設定】モニターごとに書き換えてください // ========================================== var contents = [ {type: 'video', src: 'video1_screen1.mp4', duration: 15000}, {type: 'video', src: 'video2_screen1.mp4', duration: 15000}, {type: 'image', src: 'calendar_screen1.jpg', duration: 10000} ]; var INDIVIDUAL_MS = 0; contents.forEach(function(c) { INDIVIDUAL_MS += c.duration; }); var MAIN_MS = 10000; var CYCLE_MS = INDIVIDUAL_MS + MAIN_MS; var overlay = document.getElementById('mainOverlay'); var started = false; var lastContentIdx = -1; var offsets = []; var acc = 0; contents.forEach(function(c) { offsets.push(acc); acc += c.duration; }); // 初期化：最初から動画・画像タグを裏側で作っておく var elements = contents.map(function(c) { var el = document.createElement(c.type === 'video' ? 'video' : 'div'); if (c.type === 'video') { el.src = c.src; el.muted = true; el.playsInline = true; el.preload = 'auto'; // Androidに先読みさせる el.setAttribute('muted', ''); el.setAttribute('playsinline', ''); el.setAttribute('webkit-playsinline', ''); } else { el.className = 'slide'; el.style.backgroundImage = "url('" + c.src + "')"; } document.body.appendChild(el); return el; }); // 【超重要】動画を確実に動かすための関数 function startSignage() { if (started) return; // 二重実行防止 started = true; // スタート画面を消す document.getElementById('startOverlay').style.display = 'none'; // Androidの全画面化命令（失敗しても無視して進む） try { var el = document.documentElement; if (el.requestFullscreen) el.requestFullscreen(); else if (el.webkitRequestFullscreen) el.webkitRequestFullscreen(); } catch(e) { console.log("全画面化はスキップされました"); } // 100%再生をAndroidに許可させる裏ワザ elements.forEach(function(el) { if (el.tagName === 'VIDEO') { var p = el.play(); if (p !== undefined) { p.then(function() { el.pause(); }) // 一瞬再生してすぐ止める .catch(function(e) { console.log(e); }); } } }); // 監視タイマーを起動 setInterval(tick, 200); } // 💥 スマホの「タッチ」とPCの「クリック」両方に対応させる var startBtn = document.getElementById('startOverlay'); i >= 0; i--) { if (offset >= offsets[i]) return i; } return 0; } function tick() { var offset = Date.now() % CYCLE_MS; var isMain = offset >= INDIVIDUAL_MS; if (isMain) { if (lastContentIdx !== -99) { lastContentIdx = -99; }); } } else { el.style.opacity = '0'; if (el.tagName === 'VIDEO') el.pause(); } }); } } }
